#include <iostream>
#include <string>
#include <cmath>
#include <cstdlib>
#include <ctime>

using namespace std;

enum enGameChoice { Stone = 1, Paper = 2, Scessor = 3 };
enum enWinner { Player1 = 1, Computer = 2, Draw = 3 };

struct stRoundInfo
{
    short RoundNumber = 0;
    enGameChoice Player1Choice;
    enGameChoice ComputerChoice;
    enGameChoice DrawChoice;
    enWinner Winner;
    string WinnerName;
};

struct stGameResults
{
    short Gamerounds = 0;
    short Player1WinTimes = 0;
    short ComputerWinTimes = 0;
    short DrawTimes = 0;
    enWinner GameWinner;
    string WinnerName = "";
};

int RandomNumber(int From, int To)
{
    return rand() % (To - From + 1) + From;
}

string WinnerNAme(enWinner Winner)
{
    string arrWinnerNames[3] = { "Player1", "Computer", "Draw" };
    return arrWinnerNames[Winner - 1];
}

string ChoiceName(enGameChoice Choice)
{
    string arrGameChoiceNames[3] = { "Stone", "Paper", "Scessor" };
    return arrGameChoiceNames[Choice - 1];
}

void SetWinnerScreenColor(enWinner Winner)
{
    switch (Winner)
    {
    case enWinner::Player1:
        system("Color 2F");
        break;
    case enWinner::Computer:
        system("Color 4F");
        break;
    case enWinner::Draw:
        system("Color 6F");
        break;
    }
}

void PrintRoundResult(stRoundInfo RoundInfo)
{
    cout << "\nRound [" << RoundInfo.RoundNumber << "] Result: " << endl;
    cout << " Player1 Choice: " << ChoiceName(RoundInfo.Player1Choice) << endl;
    cout << " Computer Choice: " << ChoiceName(RoundInfo.ComputerChoice) << endl;
    cout << " Round Winner: " << RoundInfo.WinnerName << endl;
    cout << "----------------------------------------" << endl;

    SetWinnerScreenColor(RoundInfo.Winner);
}

enWinner WhoWonTheGame(short Player1WinTimes, short ComputerWinTimes)
{
    if (Player1WinTimes > ComputerWinTimes)
        return enWinner::Player1;
    else if (Player1WinTimes < ComputerWinTimes)
        return enWinner::Computer;
    else
        return enWinner::Draw;
}

enWinner WhoWonTheRound(stRoundInfo RoundInfo)
{
    if (RoundInfo.Player1Choice == RoundInfo.ComputerChoice)
        return enWinner::Draw;

    switch (RoundInfo.Player1Choice)
    {
    case enGameChoice::Stone:
        return (RoundInfo.ComputerChoice == enGameChoice::Paper) ? enWinner::Computer : enWinner::Player1;
    case enGameChoice::Paper:
        return (RoundInfo.ComputerChoice == enGameChoice::Scessor) ? enWinner::Computer : enWinner::Player1;
    case enGameChoice::Scessor:
        return (RoundInfo.ComputerChoice == enGameChoice::Stone) ? enWinner::Computer : enWinner::Player1;
    }
    return enWinner::Draw;
}

stGameResults FillGameResults(short Gamerounds, short Player1WinTimes, short ComputerWinTimes, enWinner GameWinner, string WinnerName)
{
    stGameResults GameResults;
    GameResults.Gamerounds = Gamerounds;
    GameResults.Player1WinTimes = Player1WinTimes;
    GameResults.ComputerWinTimes = ComputerWinTimes;
    GameResults.DrawTimes = Gamerounds - (Player1WinTimes + ComputerWinTimes);
    GameResults.GameWinner = GameWinner;
    GameResults.WinnerName = WinnerName;
    return GameResults;
}

enGameChoice ReadPlayer1Choice()
{
    short Choice;
    do
    {
        cout << "\nYour Choice: [1] Stone, [2] Paper, [3] Scessor ? ";
        cin >> Choice;
    } while (Choice < 1 || Choice > 3);
    return (enGameChoice)Choice;
}

enGameChoice GetComputerChoice()
{
    return (enGameChoice)RandomNumber(1, 3);
}

stGameResults PlayGame(short HowManyRounds)
{
    stRoundInfo RoundInfo;
    short Player1WinTimes = 0, ComputerWinTimes = 0, DrawTimes = 0;

    for (short GameRound = 1; GameRound <= HowManyRounds; GameRound++)
    {
        RoundInfo.RoundNumber = GameRound;
        RoundInfo.Player1Choice = ReadPlayer1Choice();
        RoundInfo.ComputerChoice = GetComputerChoice();
        RoundInfo.Winner = WhoWonTheRound(RoundInfo);
        RoundInfo.WinnerName = WinnerNAme(RoundInfo.Winner);

        if (RoundInfo.Winner == enWinner::Player1)
            Player1WinTimes++;
        else if (RoundInfo.Winner == enWinner::Computer)
            ComputerWinTimes++;
        else
            DrawTimes++;

        PrintRoundResult(RoundInfo);
    }
    enWinner FinalWinner = WhoWonTheGame(Player1WinTimes, ComputerWinTimes);
    return FillGameResults(HowManyRounds, Player1WinTimes, ComputerWinTimes, FinalWinner, WinnerNAme(FinalWinner));
}

string Tab(short HowManyOfTabs)
{
    string Tabs = "";
    for (short i = 0; i < HowManyOfTabs; i++)
        Tabs += "\t";
    return Tabs;
}

void ShowGameOerScreen()
{
    cout << Tab(2) << "_____________________________________________________________" << endl;
    cout << Tab(2) << "==================== Game Over ====================" << endl;
    cout << Tab(2) << "_____________________________________________________________" << endl;
}

void ShowFinalGameResults(stGameResults GameResults)
{
    cout << Tab(2) << "_____________________[Final Game Results]___________________ " << endl;
    cout << Tab(2) << "Total Rounds: " << GameResults.Gamerounds << endl;
    cout << Tab(2) << "Player1 Win Times: " << GameResults.Player1WinTimes << endl;
    cout << Tab(2) << "Computer Win Times: " << GameResults.ComputerWinTimes << endl;
    cout << Tab(2) << "Draw Times: " << GameResults.DrawTimes << endl;
    cout << Tab(2) << "Game Winner: " << GameResults.WinnerName << endl;
    cout << Tab(2) << "_____________________________________________________________" << endl;
}

short ReadHowManyRounds()
{
    short GameRounds;
    do
    {
        cout << "How Many Rounds 1 To 10? " << endl;
        cin >> GameRounds;
    } while (GameRounds < 1 || GameRounds > 10);
    return GameRounds;
}

void ResetScreen()
{
    system("cls");
    system("Color 0F");
}

void StartGame()
{
    char PlayAgain = 'Y';

    do
    {
        ResetScreen();
        stGameResults GameResults = PlayGame(ReadHowManyRounds());
        ShowGameOerScreen();
        ShowFinalGameResults(GameResults);

        cout << "Do You Want To Play Again? Y/N? ";
        cin >> PlayAgain;

    } while (PlayAgain == 'Y' || PlayAgain == 'y');
}

int main()
{
    srand((unsigned)time(NULL));

    StartGame();
    return 0;
}
